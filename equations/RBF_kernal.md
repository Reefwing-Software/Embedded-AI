<!-- RBF Kernel Function -->
$$
\text{RBF Kernel Function: }
k(x, x') = \exp\left(-\frac{(x - x')^2}{2 \ell^2}\right)
$$

<br>

<!-- Covariance Matrix -->
$$
\text{Covariance Matrix: }
K_{ij} = k(x_i, x_j)
$$
